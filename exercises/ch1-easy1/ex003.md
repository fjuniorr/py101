def main(start=1, end=99):
    """
    Print even numbers from start to end (inclusive).

    Details
    ===
    It iterates only over even numbers
    """
    if start % 2 == 1:
        start += 1
    
    for n in range(start, end+1, 2):
        print(n)

main()
